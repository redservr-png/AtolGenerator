================================================================
ВНЕШНЯЯ ОБРАБОТКА 1С: «Импорт ФПД/№ФД в реализации из CSV»
================================================================
Зачем нужна: программа AtolGenerator при сопоставлении XML+ОФД сохраняет
рядом со своим .exe файл вида  atol_to_1c_YYYYMMDD_HHmmss.csv  с данными:
  НомерРеализации;ФПД;НомерФД;ДатаПечатиЧека

Если запись через COM падает (NullReferenceException при ПолучитьОбъект),
этой обработкой можно загрузить тот же CSV ВНУТРИ 1С — без COM, без NRE.

================================================================
КАК СОЗДАТЬ ОБРАБОТКУ
================================================================
1. Открыть Конфигуратор 1С (или режим разработки в УТ 10.3)
2. Файл → Новый → Внешняя обработка
3. Имя:           ИмпортЧекНомерФПИзCSV
4. Открыть «Формы» → Добавить → Форма
5. На форме добавить:
   - Реквизит формы:           ПутьКФайлу (Строка, длина 255)
   - Поле:                     ПутьКФайлу (с кнопкой выбора файла)
   - Реквизит:                 ПерезаписыватьЗаполненные (Булево)
   - Флажок:                   ПерезаписыватьЗаполненные
   - Кнопка:                   ВыполнитьИмпорт
6. В модуль формы вставить код ниже (после &НаКлиенте)
7. Сохранить как файл .epf куда удобно

================================================================
КОД МОДУЛЯ ФОРМЫ (вставить целиком)
================================================================

&НаКлиенте
Процедура ПутьКФайлуНачалоВыбора(Элемент, ДанныеВыбора, СтандартнаяОбработка)
    СтандартнаяОбработка = Ложь;
    Диалог = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
    Диалог.Заголовок = "Выберите CSV-файл от AtolGenerator";
    Диалог.Фильтр    = "CSV-файлы (*.csv)|*.csv|Все файлы (*.*)|*.*";
    Если Диалог.Выбрать() Тогда
        ПутьКФайлу = Диалог.ПолноеИмяФайла;
    КонецЕсли;
КонецПроцедуры

&НаКлиенте
Процедура ВыполнитьИмпорт(Команда)
    Если ПустаяСтрока(ПутьКФайлу) Тогда
        Сообщить("Выберите CSV-файл.");
        Возврат;
    КонецЕсли;
    Результат = ИмпортироватьНаСервере(ПутьКФайлу, ПерезаписыватьЗаполненные);
    ПоказатьПредупреждение(, Результат);
КонецПроцедуры

&НаСервере
Функция ИмпортироватьНаСервере(ПутьФайла, Перезаписывать)
    Текст = Новый ЧтениеТекста(ПутьФайла, КодировкаТекста.UTF8);
    Обновлено  = 0;
    Пропущено  = 0;
    НеНайдено  = 0;
    Ошибок     = 0;
    НомерСтроки = 0;
    Лог = Новый Массив;

    Стр = Текст.ПрочитатьСтроку(); // заголовок
    Пока Истина Цикл
        Стр = Текст.ПрочитатьСтроку();
        Если Стр = Неопределено Тогда
            Прервать;
        КонецЕсли;
        НомерСтроки = НомерСтроки + 1;

        Поля = СтрРазделить(Стр, ";", Истина);
        Если Поля.Количество() < 4 Тогда
            Продолжить;
        КонецЕсли;

        НомерРеализации = СокрЛП(Поля[0]);
        ФПД             = СокрЛП(Поля[1]);
        НомерФД         = СокрЛП(Поля[2]);
        ДатаЧекаСтр     = СокрЛП(Поля[3]);

        Если ПустаяСтрока(НомерРеализации) ИЛИ ПустаяСтрока(ФПД) Тогда
            Продолжить;
        КонецЕсли;

        Попытка
            ДатаЧека = НачалоДня(ТекущаяДата());
            // Формат ДатаПечатиЧека: "дд.мм.гггг чч:мм:сс"
            ВремФрагменты = СтрРазделить(ДатаЧекаСтр, " ", Истина);
            Если ВремФрагменты.Количество() >= 1 Тогда
                ЧастиДаты = СтрРазделить(ВремФрагменты[0], ".", Истина);
                Если ЧастиДаты.Количество() = 3 Тогда
                    ДД = Число(ЧастиДаты[0]);
                    ММ = Число(ЧастиДаты[1]);
                    ГГ = Число(ЧастиДаты[2]);
                    Чч = 0; Мн = 0; Сек = 0;
                    Если ВремФрагменты.Количество() >= 2 Тогда
                        ЧастиВр = СтрРазделить(ВремФрагменты[1], ":", Истина);
                        Если ЧастиВр.Количество() >= 1 Тогда Чч = Число(ЧастиВр[0]); КонецЕсли;
                        Если ЧастиВр.Количество() >= 2 Тогда Мн = Число(ЧастиВр[1]); КонецЕсли;
                        Если ЧастиВр.Количество() >= 3 Тогда Сек = Число(ЧастиВр[2]); КонецЕсли;
                    КонецЕсли;
                    ДатаЧека = Дата(ГГ, ММ, ДД, Чч, Мн, Сек);
                КонецЕсли;
            КонецЕсли;

            // Ищем самый свежий документ с этим номером, не позднее даты чека
            Запрос = Новый Запрос("
            |ВЫБРАТЬ ПЕРВЫЕ 1
            |    Док.Ссылка       КАК Ссылка,
            |    Док.ЧекНомерФП   КАК ЧекНомерФП,
            |    Док.Комментарий  КАК Комментарий
            |ИЗ
            |    Документ.РеализацияТоваровУслуг КАК Док
            |ГДЕ
            |    Док.Номер = &Номер
            |    И Док.ПометкаУдаления = ЛОЖЬ
            |    И Док.Дата <= &ДатаЧека
            |УПОРЯДОЧИТЬ ПО
            |    Док.Дата УБЫВ
            |");
            Запрос.УстановитьПараметр("Номер",    НомерРеализации);
            Запрос.УстановитьПараметр("ДатаЧека", ДатаЧека);
            Выборка = Запрос.Выполнить().Выбрать();

            Если НЕ Выборка.Следующий() Тогда
                НеНайдено = НеНайдено + 1;
                Лог.Добавить(НомерРеализации + ": документ не найден");
                Продолжить;
            КонецЕсли;

            ТекущийФП = СокрЛП("" + Выборка.ЧекНомерФП);
            Если НЕ Перезаписывать И НЕ ПустаяСтрока(ТекущийФП) И ТекущийФП <> "0" И ТекущийФП <> "999999999" Тогда
                Пропущено = Пропущено + 1;
                Продолжить;
            КонецЕсли;

            ОбъектДок = Выборка.Ссылка.ПолучитьОбъект();
            ОбъектДок.ЧекНомерФП     = ФПД;
            ОбъектДок.НомерЧекаККМ   = НомерФД;
            ОбъектДок.ДатаПечатиЧека = ДатаЧека;

            // Комментарий: пустой → marker; непустой → дописываем через "   \\\   "
            Маркер = "Пробит чек коррекции ""Приход""";
            ТекущКомментарий = СокрЛП(Выборка.Комментарий);
            Если Найти(ТекущКомментарий, "Пробит чек коррекции") = 0 Тогда
                Если ПустаяСтрока(ТекущКомментарий) Тогда
                    ОбъектДок.Комментарий = Маркер;
                Иначе
                    ОбъектДок.Комментарий = ТекущКомментарий + "   \\\   " + Маркер;
                КонецЕсли;
            КонецЕсли;

            ОбъектДок.Записать();
            Обновлено = Обновлено + 1;
        Исключение
            Ошибок = Ошибок + 1;
            Лог.Добавить(НомерРеализации + ": " + ОписаниеОшибки());
        КонецПопытки;
    КонецЦикла;
    Текст.Закрыть();

    Сообщение = "Обновлено: " + Обновлено + Символы.ПС +
                "Пропущено: " + Пропущено + Символы.ПС +
                "Не найдено: " + НеНайдено + Символы.ПС +
                "Ошибок: " + Ошибок;

    Если Лог.Количество() > 0 Тогда
        Сообщение = Сообщение + Символы.ПС + Символы.ПС + "Первые проблемы:";
        Для Сч = 0 По Мин(9, Лог.Количество() - 1) Цикл
            Сообщение = Сообщение + Символы.ПС + Лог[Сч];
        КонецЦикла;
    КонецЕсли;

    Возврат Сообщение;
КонецФункции

================================================================
КАК ИСПОЛЬЗОВАТЬ
================================================================
1. Запустить AtolGenerator → нажать «Сопоставить XML+ОФД → 1С»
2. После работы в окне результата программа покажет путь к CSV-файлу,
   например: C:\AtolGenerator\atol_to_1c_20260521_103045.csv
3. Открыть 1С (УТ 10.3) → Файл → Открыть → выбрать вашу .epf обработку
4. В форме обработки:
   - В поле «ПутьКФайлу» выбрать тот самый CSV
   - Снять флажок «Перезаписывать заполненные» (рекомендуется)
   - Нажать «Выполнить импорт»
5. Дождаться сообщения «Обновлено: NN, Пропущено: KK, ...»

ПРЕИМУЩЕСТВА перед COM:
- Работает ВНУТРИ 1С → нет COM-marshalling → нет NRE
- Соблюдает все права и блокировки штатно
- Использует ту же логику что и групповая обработка, которая у вас работает
